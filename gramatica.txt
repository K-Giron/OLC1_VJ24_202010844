<INICIO> ::= <INSTRUCCIONES>

<INSTRUCCIONES> ::= <INSTRUCCIONES> <INSTRUCCION>
                  | <INSTRUCCION>

<INSTRUCCION> ::= <IMPRESION>
                | <DECLARACION>
                | <ASIGNACION>
                | <IIF>
                | <IFOR>
                | <BREAKK>
                | <CONTINUEE>
                | <INCREMENTO> FINCADENA
                | <DECREMENTO> FINCADENA
                | <IWHILE>
                | <IDOWHILE>
                | <IMATCH>
                | error FINCADENA

<IMPRESION> ::= IMPRIMIR PAR1 <EXPRESION> PAR2 FINCADENA

<DECLARACION> ::= MUTABILIDAD ID DOSPUNTOS <TIPOS> IGUAL <EXPRESION> FINCADENA
                | MUTABILIDAD ID DOSPUNTOS <TIPOS> FINCADENA

<ASIGNACION> ::= ID IGUAL <EXPRESION> FINCADENA

<IIF> ::= IF PAR1 <EXPRESION> PAR2 LLAVE1 <INSTRUCCIONES> LLAVE2
         | ELSE LLAVE1 <INSTRUCCIONES> LLAVE2
         | ELSE <IIF>

<IFOR> ::= FOR PAR1 <ASIGNACION> <EXPRESION> FINCADENA <ACTFOR> PAR2 LLAVE1 <INSTRUCCIONES> LLAVE2

<ACTFOR> ::= ID IGUAL <EXPRESION>
           | <INCREMENTO>
           | <DECREMENTO>

<BREAKK> ::= BREAK FINCADENA

<CONTINUEE> ::= CONTINUE FINCADENA

<INCREMENTO> ::= ID MAS MAS

<DECREMENTO> ::= ID MENOS MENOS

<TIPOS> ::= INT
          | DOUBLE
          | STRING
          | CHAR
          | BOOL

<IWHILE> ::= WHILE PAR1 <EXPRESION> PAR2 LLAVE1 <INSTRUCCIONES> LLAVE2

<IDOWHILE> ::= DO LLAVE1 <INSTRUCCIONES> LLAVE2 WHILE PAR1 <EXPRESION> PAR2 FINCADENA

<IMATCH> ::= MATCH <EXPRESION> LLAVE1 <INSTRUCCIONES> LLAVE2

<EXPRESION> ::= MENOS <EXPRESION>
              | <EXPRESION> MAS <EXPRESION>
              | <EXPRESION> MENOS <EXPRESION>
              | <EXPRESION> MULTIPLICACION <EXPRESION>
              | <EXPRESION> DIVISION <EXPRESION>
              | <EXPRESION> POTENCIA <EXPRESION>
              | <EXPRESION> MODULO <EXPRESION>
              | <EXPRESION> IGUALACION <EXPRESION>
              | <EXPRESION> MENORQUE <EXPRESION>
              | <EXPRESION> MAYORQUE <EXPRESION>
              | <EXPRESION> MENOR <EXPRESION>
              | <EXPRESION> MAYOR <EXPRESION>
              | <EXPRESION> DIFERENTE <EXPRESION>
              | <EXPRESION> OR <EXPRESION>
              | <EXPRESION> AND <EXPRESION>
              | <EXPRESION> XOR <EXPRESION>
              | NOT <EXPRESION>
              | PAR1 <TIPOS> PAR2 <EXPRESION>
              | ENTERO
              | DECIMAL
              | CADENA
              | BOOLEANO
              | CARACTER
              | ID
              | PAR1 <EXPRESION> PAR2
